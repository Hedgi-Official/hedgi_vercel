✅ Mercado Pago Brick Integration: Full Specification
🔧 Goal
Build a Mercado Pago payment flow using the Brick SDK that:

Accepts payment via an embedded card form (cardPayment Brick)

Processes the payment on your Flask backend

Displays a success or failure status screen (statusScreen Brick)

On success, places a trade using a refund token (payment.id) returned from Mercado Pago

🧭 Overview of Endpoints
Endpoint	Method	Description
/payment	GET	Renders HTML Brick for card input
/process_payment	POST	Handles submission from card Brick; charges card
/payment_status/<payment_id>	GET	Displays payment status using statusScreen Brick
/api/trades	POST	Called upon successful payment to place hedge

1️⃣ /payment?amount=<AMOUNT>
Purpose: Renders HTML with embedded cardPayment Brick.

The amount query param sets the transaction value.

The HTML should look like:

html
Copy
Edit
<html>
<head>
  <script src="https://sdk.mercadopago.com/js/v2"></script>
</head>
<body>
  <div id="cardPaymentBrick_container"></div>
  <script>
    const mp = new MercadoPago('YOUR_PUBLIC_KEY', { locale: 'en-US' });
    const bricksBuilder = mp.bricks();

    const renderCardPaymentBrick = async () => {
      const settings = {
        initialization: {
          amount: AMOUNT_PLACEHOLDER,
        },
        customization: {
          paymentMethods: {
            ticket: 'all',
            bankTransfer: 'all',
            creditCard: 'all',
          },
        },
        callbacks: {
          onReady: () => {},
          onSubmit: ({ selectedPaymentMethod, formData }) => {
            return fetch('/process_payment', {
              method: 'POST',
              headers: { 'Content-Type': 'application/json' },
              body: JSON.stringify(formData)
            })
            .then(res => res.json())
            .then(data => {
              if (data.status === 'approved') {
                window.location.href = `/payment_status/${data.id}`;
              } else {
                window.location.href = `/payment_status/failure`;
              }
            });
          },
          onError: error => console.error('Brick error', error),
        },
      };

      window.cardPaymentBrickController = await bricksBuilder.create('cardPayment', 'cardPaymentBrick_container', settings);
    };

    renderCardPaymentBrick();
  </script>
</body>
</html>

3️⃣ /payment_status/<payment_id> (GET)
Purpose: Displays a visual confirmation using statusScreen Brick.

Success HTML (payment_status.html):

html
Copy
Edit
<html>
<head>
  <script src="https://sdk.mercadopago.com/js/v2"></script>
</head>
<body>
  <div id="statusScreenBrick_container"></div>
  <script>
    const mp = new MercadoPago('YOUR_PUBLIC_KEY', { locale: 'en-US' });
    const bricksBuilder = mp.bricks();

    const renderStatusScreenBrick = async () => {
      const settings = {
        initialization: { paymentId: 'PAYMENT_ID_PLACEHOLDER' },
        customization: {
          visual: {
            hideStatusDetails: true,
            hideTransactionDate: true,
            style: { theme: 'default' },
          },
          backUrls: {
            error: '/error',
            return: '/homepage',
          },
        },
        callbacks: {
          onReady: async () => {
            // Auto-place trade
            await fetch('/api/trades', {
              method: 'POST',
              headers: { 'Content-Type': 'application/json' },
              body: JSON.stringify({
                symbol: "EURUSD",
                volume: 0.1,
                direction: "buy",
                metadata: {
                  days: 7,
                  paymentToken: 'PAYMENT_ID_PLACEHOLDER',
                  margin: 500
                }
              })
            });
          },
          onError: (error) => console.error('Status screen error', error),
        },
      };
      await bricksBuilder.create('statusScreen', 'statusScreenBrick_container', settings);
    };

    renderStatusScreenBrick();
  </script>
</body>
</html>
