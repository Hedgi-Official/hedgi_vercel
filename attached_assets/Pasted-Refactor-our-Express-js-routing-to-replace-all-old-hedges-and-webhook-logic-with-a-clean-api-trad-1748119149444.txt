//    Proxies to HTTP POST ${FLASK}/trades
//    Body must be { symbol, direction, volume, status:'NEW', metadata }
//    Persist the returned trade in your DB with status='NEW'
app.post('/api/trades', …);

// 2) Poll a trade’s status → GET /api/trades/:id/status
//    Proxies to HTTP GET ${FLASK}/trades/:id/status
//    Returns { status: 'NEW'|'Executed'|'Closed'|'FAILED' }
//    Map for UI:
//      NEW      → label 'Order sent'   (still Active)
//      Executed → label 'Order placed' (still Active)
//      Closed   → move to History
//      FAILED   → move to History with 'Failed' marker
//    Update your DB record’s status & updatedAt, then return { status, label }
app.get('/api/trades/:id/status', …);

// 3) Close a trade early → POST /api/trades/:id/close
//    Proxies to HTTP POST ${FLASK}/trades/:id/close
//    Return the raw Flask JSON; client will poll /status until it becomes Closed or FAILED
app.post('/api/trades/:id/close', …);

// 4) History tab → GET /api/trades/history
