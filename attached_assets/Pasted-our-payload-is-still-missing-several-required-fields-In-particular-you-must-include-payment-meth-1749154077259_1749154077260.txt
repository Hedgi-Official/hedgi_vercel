payment_method_id (e.g. "master", "visa").

transaction_amount, not "amount".

description (a non‐empty string).

payer.first_name (cannot be blank).

Below is exactly how your iframe’s onSubmit should build the JSON body. Copy this snippet into your payment_BRL.html, replacing whatever you currently have:

<script>
  // …inside onSubmit(cardFormData)…
  return fetch("/api/proxy/process_payment", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      token:               cardFormData.token,               // already correct
      installments:        cardFormData.installments,        // already correct
      payment_method_id:   cardFormData.paymentMethodId,     // ← REQUIRED
      
      // Rename "amount" → "transaction_amount":
      transaction_amount:  cardFormData.transactionAmount,    // ← REQUIRED
      
      // Provide a non‐empty description:
      description:         `Hedgi order for R$${cardFormData.transactionAmount}`,
      
      payer: {
        email:            cardFormData.payer.email,           // already correct
        identification: {
          type:           cardFormData.payer.identification?.type   || "CPF",
          number:         cardFormData.payer.identification?.number || ""
        },
        // Make sure first_name is not empty:
        first_name:       cardFormData.payer.first_name || "HedgiCustomer"
      },
      txId:                TX_ID                             // already correct
    })
  })
  .then(async (res) => {
    if (!res.ok) {
      const text = await res.text();
      window.parent.postMessage({ status: "error", txId: TX_ID, error: text }, "*");
      throw new Error(text);
    }
    const json = await res.json();
    if (json.status === "approved") {
      window.parent.postMessage({ status: "approved", txId: TX_ID, data: json }, "*");
    } else {
      window.parent.postMessage({ status: "error", txId: TX_ID, error: json.error || "Not approved" }, "*");
    }
  })
  .catch((err) => {
    window.parent.postMessage({ status: "error", txId: TX_ID, error: err.message }, "*");
  });
</script>

What’s still missing in your payload:

You used "amount": 415. Flask expects "transaction_amount": 415.

There is no "payment_method_id" field at all—write payment_method_id: cardFormData.paymentMethodId.

You have no "description"—it must be a non‐empty string like "Hedgi order for R$415".

Inside payer, there is no first_name. You must supply a fallback (e.g. "HedgiCustomer") if the Brick doesn’t supply one.
